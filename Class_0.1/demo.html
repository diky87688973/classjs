// Dog.js
import Animal;    // 引入依赖
import IMyItface;

class Dog extends Animal implements IMyItface {
    
    var a = 'aaa';          // 私有成员
    this.name = 'dog';      // 公有成员, 实例通过 this.name 访问
    
    Dog( age, sex, name ) {
        super( age, sex );
        
        this.name = name;
    }
    
    say( msg ) {
        return this.name + ',' + super( msg );   // super(...) 调用父类同名方法
    }
    
    method1() {
        return 'method1 call ' + IMyItface.PROPERTY1;
    }
      
    method2() {
        return 'method2 call ' + IMyItface.PROPERTY2;
    }
      
}

// Animal.js
class Animal {

    this.age = 0;
    this.sex = '未知';
    
    Animal( age, sex ) {
        super();        // 调用父类构造方法

        this.age = age;
        this.sex = sex;
    }

    say( msg ) {
        return this.age + ',' + this.sex + ',say:' + msg;
    }
}

// ---- 闪瞎双眼的分割线 ---

// IMyItface.js
interface IMyItface {
    method1();      // 接口方法定义
    method2();
    this.PROPERTY1 = '接口中的静态常量';
    this.PROPERTY2 = true;
}
编译后的 Class.js 代码 
// Dog.js
Import( "Animal" );    // 引入依赖
Import( "IMyItface" );

Class( "Dog", Animal, IMyItface, function(){
    
    var a = 'aaa';          // 私有成员
    this.name = 'dog';      // 公有成员, 实例通过 this.name 访问
    
    this.Dog=function( age, sex, name ) {
        Super( age, sex );
        
        this.name = name;
    }
    
    this.say=function( msg ) {
        return this.name + ',' + Super( msg );   // super(...) 调用父类同名方法
    }
    
    this.method1=function() {
        return 'method1 call ' + IMyItface.PROPERTY1;
    }
      
    this.method2=function() {
        return 'method2 call ' + IMyItface.PROPERTY2;
    }
      
});

// ---- 闪瞎双眼的分割线 ---

// Animal.js
Class( "Animal", function(){

    this.age = 0;
    this.sex = '未知';
    
    this.Animal=function( age, sex ) {
        Super();        // 调用父类构造方法

        this.age = age;
        this.sex = sex;
    }

    this.say=function( msg ) {
        return this.age + ',' + this.sex + ',say:' + msg;
    }
});

// ---- 闪瞎双眼的分割线 ---

// IMyItface.js
Interface( "IMyItface", function(){
    this.method1=Function;      // 接口方法定义
    this.method2=Function;
    this.PROPERTY1 = '接口中的静态常量';
    this.PROPERTY2 = true;
});
